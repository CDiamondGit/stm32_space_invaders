void initGameState(GameState* gs) {
  gs->ship.coords.x = 58;
  gs->ship.coords.y = 130;
  gs->ship.coords.oldX = 58;
  gs->ship.coords.oldY = 130;
  gs->ship.shipSpeed = 5;

  gs->bullet.coords.x = gs->ship.coords.x + BULLET_OFFSET_X;
  gs->bullet.coords.y = gs->ship.coords.y - BULLET_OFFSET_Y;
  gs->bullet.coords.oldX = gs->ship.coords.x + BULLET_OFFSET_X;
  gs->bullet.coords.oldY = gs->ship.coords.y - BULLET_OFFSET_Y;
  gs->bullet.bulletSpeed = 4;
  gs->bullet.bulletState = READY;
}

GameState gs;
uint32_t lastUpdate = 0;
uint32_t frameDelay = 16;

int main() {
  initClock();
  initSysTick();
  setupIO();
  // initSound();
  initGameState(&gs);
  // twinkle_twinkle();

  for (int i = 0; i < alienRow; i++) {
    uint16_t posY = (i * (8 + 5) + 20);
    for (int j = 0; j < alienColumn; j++) {
      uint16_t posX = j * 22 + 12;
      if (alienStatus[i][j] == 0) {
        putImage(posX, posY, 11, 8, blueAlien, 1, 0);  // alien sprite
      }
    }
  }

  // initial sprie render
  putImage(gs.ship.coords.x, gs.ship.coords.y, 12, 12, spaceShip, 1, 0);

  drawLine(0, 144, 128, 144, 57351);

  // game loop
  while (1) {
    uint32_t now = milliseconds;

    // TODO FPS implementation
    if (now - lastUpdate >= frameDelay) {
      lastUpdate = now;

      // if right button is pressed, move ship to right
      if ((GPIOB->IDR & (1 << 4)) == 0) {
        if (gs.ship.coords.x < (128 - 20)) {
          gs.ship.coords.x += gs.ship.shipSpeed;
          printNumber(gs.ship.coords.x, 10, 100, 1, 0);
        }
      }

      // if left button is pressed move ship to left;
      if ((GPIOB->IDR & (1 << 5)) == 0) {
        if (gs.ship.coords.x > 12) {
          gs.ship.coords.x -= gs.ship.shipSpeed;
          printNumber(gs.ship.coords.x, 10, 100, 1, 0);
        }
      }

      // if up button is pressed render bullet;
      if ((GPIOA->IDR & (1 << 8)) == 0) {
        if (gs.bullet.coords.y > 2 && gs.bullet.bulletState == READY) {
          gs.bullet.bulletState = FIRE;
          // render_bullet(&gs.bullet);
        }
      }

      if (gs.alien.isChanged == 1) {
        for (int i = 0; i < alienRow; i++) {
          uint16_t posY = (i * (8 + 5) + 20);
          for (int j = 0; j < alienColumn; j++) {
            uint16_t posX = j * 22 + 12;
            if (alienStatus[i][j] == 0) {
              putImage(posX, posY, 11, 8, blueAlien, 1, 0);  // alien sprite
            }
          }
        }
      }

      // if ship position changed, then render ship
      if ((gs.ship.coords.oldX != gs.ship.coords.x) ||
          gs.ship.coords.oldY != gs.ship.coords.y) {
        fillRectangle(gs.ship.coords.oldX, gs.ship.coords.oldY, 12, 12, 0);
        gs.ship.coords.oldX = gs.ship.coords.x;
        gs.ship.coords.oldY = gs.ship.coords.y;
        putImage(gs.ship.coords.x, gs.ship.coords.y, 12, 12, spaceShip, 1, 0);
        if (gs.bullet.bulletState != FIRE) {
        }
      }

      updateCollision();

      // if bullet fired, render bullet;
      if (gs.bullet.bulletState == FIRE) {
        gs.bullet.coords.y -= gs.bullet.bulletSpeed;

        if (gs.bullet.coords.y < 5) {
          fillRectangle(gs.bullet.coords.oldX, gs.bullet.coords.oldY, 2, 2, 0);
          gs.bullet.coords.x = gs.bullet.coords.oldX =
              gs.ship.coords.x + BULLET_OFFSET_X;
          gs.bullet.coords.y = gs.bullet.coords.oldY =
              gs.ship.coords.y + BULLET_OFFSET_Y;
          gs.bullet.bulletState = READY;
        }
        render_bullet(&gs.bullet);
      } else {
        gs.bullet.coords.x = gs.ship.coords.x + BULLET_OFFSET_X;
        gs.bullet.coords.y = gs.ship.coords.y - BULLET_OFFSET_Y;
      }
      gs.alien.isChanged = 0;
      // printNumber(gs.bullet.bulletState, 10, 10, 1, 0);
      // delay(50);
    }
  }
  return 0;
}

int checkCollision(uint16_t o1X,
                   uint16_t o1Y,
                   uint16_t o1w,
                   uint16_t o1h,
                   uint16_t o2X,
                   uint16_t o2Y,
                   uint16_t o2w,
                   uint16_t o2h) {
  if (o1X + o1w <= o2X)
    return 0;  // A is left  of B
  if (o1X >= o2X + o2w)
    return 0;  // A is right of B
  if (o1Y + o1h <= o2Y)
    return 0;  // A is above  B
  if (o1Y >= o2Y + o2h)
    return 0;
  return 1;
}

void handle_alien_bullet_collision(Alien* a, Bullet* B) {
  for (int i = 0; i < alienRow; i++) {
    uint16_t posY = (i * (8 + 5) + 20);  // i * (height+gap) + offsetFromTop
    for (int j = 0; j < alienColumn; j++) {
      uint16_t posX =
          (j * (11 + 10) + 12);  // j * (width+gap) + offsetFrom Left
      if (checkCollision(a->coords.x + posX, a->coords.y + posY, 11, 8,
                         B->coords.x, B->coords.y, 2, 2)) {
        printText("hit", 10, 10, 1, 1);
        B->bulletSpeed = READY;
        alienStatus[i][j] = 1;
        a->isChanged = 1;
      }
    }
  }
}

void updateCollision() {
  handle_alien_bullet_collision(&gs.alien, &gs.bullet);
}

void initSysTick(void) {
  SysTick->LOAD = 48000;
  SysTick->CTRL = 7;
  SysTick->VAL = 10;
  __asm(" cpsie i ");  // enable interrupts
}
void SysTick_Handler(void) {
  milliseconds++;
}
void initClock(void) {
  // This is potentially a dangerous function as it could
  // result in a system with an invalid clock signal - result: a stuck system
  // Set the PLL up
  // First ensure PLL is disabled
  RCC->CR &= ~(1u << 24);
  while ((RCC->CR & (1 << 25)))
    ;  // wait for PLL ready to be cleared

  // Warning here: if system clock is greater than 24MHz then wait-state(s)
  // need to be inserted into Flash memory interface

  FLASH->ACR |= (1 << 0);
  FLASH->ACR &= ~((1u << 2) | (1u << 1));
  // Turn on FLASH prefetch buffer
  FLASH->ACR |= (1 << 4);
  // set PLL multiplier to 12 (yielding 48MHz)
  RCC->CFGR &= ~((1u << 21) | (1u << 20) | (1u << 19) | (1u << 18));
  RCC->CFGR |= ((1 << 21) | (1 << 19));

  // Need to limit ADC clock to below 14MHz so will change ADC prescaler to 4
  RCC->CFGR |= (1 << 14);

  // and turn the PLL back on again
  RCC->CR |= (1 << 24);
  // set PLL as system clock source
  RCC->CFGR |= (1 << 1);
}
void delay(volatile uint32_t dly) {
  uint32_t end_time = dly + milliseconds;
  while (milliseconds != end_time)
    __asm(" wfi ");  // sleep
}

void enablePullUp(GPIO_TypeDef* Port, uint32_t BitNumber) {
  Port->PUPDR =
      Port->PUPDR & ~(3u << BitNumber * 2);  // clear pull-up resistor bits
  Port->PUPDR = Port->PUPDR | (1u << BitNumber * 2);  // set pull-up bit
}
void pinMode(GPIO_TypeDef* Port, uint32_t BitNumber, uint32_t Mode) {
  /*
   */
  uint32_t mode_value = Port->MODER;
  Mode = Mode << (2 * BitNumber);
  mode_value = mode_value & ~(3u << (BitNumber * 2));
  mode_value = mode_value | Mode;
  Port->MODER = mode_value;
}
int isInside(uint16_t x1,
             uint16_t y1,
             uint16_t w,
             uint16_t h,
             uint16_t px,
             uint16_t py) {
  // checks to see if point px,py is within the rectange defined by x,y,w,h
  uint16_t x2, y2;
  x2 = x1 + w;
  y2 = y1 + h;
  int rvalue = 0;
  if ((px >= x1) && (px <= x2)) {
    // ok, x constraint met
    if ((py >= y1) && (py <= y2))
      rvalue = 1;
  }
  return rvalue;
}

void setupIO() {
  RCC->AHBENR |= (1 << 18) + (1 << 17);  // enable Ports A and B
  display_begin();
  pinMode(GPIOB, 4, 0);   // right button -> blue wire-> PB4
  pinMode(GPIOB, 5, 0);   // left button -> green wire-> PB5
  pinMode(GPIOA, 8, 0);   // up button -> yellow wire -> PA8
  pinMode(GPIOA, 11, 0);  // down button -> purple wire => Pa8
  enablePullUp(GPIOB, 4);
  enablePullUp(GPIOB, 5);
  enablePullUp(GPIOA, 11);
  enablePullUp(GPIOA, 8);
}
